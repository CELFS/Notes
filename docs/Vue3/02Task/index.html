  • 3）创建路由规则 + 创建路由器 index.js
    • 路由规则用于创建路由器；
    • 路由器历史具有两种模式 --> 按需选择
      • 传统模式
      • Hash模式（ /#/ 标识，与个人笔记网址后缀路径一致）

// import { createRouter, createWebHistory } from 'vue-router'
import { createRouter, createWebHashHistory } from 'vue-router'

// 创建路由规则（数组存储）
const routes = [
    {
        path: "/",
        component: () => import("../views/index.vue")
    },
    {
        path: "/content",
        component: () => import("../views/content.vue")
    },
];

// 创建路由器
const router = createRouter({

    routes
});

// 导出路由器
export default router;

02 配置路径别名@和VSCode路径提示​

01 代码流程​

    • 添加 resolve 参数
// vite.config.js
import { defineConfig } from 'vite';
import vue from '@vitejs/plugin-vue';
import path from 'path'; // 导入 node.js path

// https://vitejs.dev/config/
export default defineConfig({
  plugins: [vue()],
  resolve: {
    alias: { // 配置路径别名
      '@': path.resolve(__dirname, 'src')
    }
  }
});

// jsconfig.json
{

        "baseUrl": ".",
        "paths": {
            "@/*": [
                "src/*"
            ] // 配置 @ 符号指向 src 目录及其子目录
        }
    }
}

  • 另外，为什么在路由规则定义同名的属性 name ？

07 全局前置守卫​

01 概括​

  • 全局前置守卫（后端：前置中间件），用于对请求拦截。例如用户未登录，不得访问相应的订单页面、个人信息页面。
  • 分类
    • 全局前置守卫（常用）
    • 全局解析守卫
    • 全局后置钩子

02 经验​

  • 实际开发中，很少有前端去写权限判断，从安全的角度，控制台容易直接注入 JS，因此，一般的权限判断都是由后端进行拦截。

02 代码流程​

  • 1）跟踪路由
  • 2）设置拦截（点击链接 --> 未发生跳转 --> 被拦截）
import { createApp } from 'vue';
import App from './App.vue';

import router from './router';

// createApp(App).use(router).mount('#app');
const app = createApp(App);
app.use(router);

// 全局前置守卫
router.beforeEach((to, from, next) => {
    console.log("to:", to);     // 即将进入的路由信息
    console.log("from", from);  // 即将离开的路由信息

    // next();
    if (to.name == "member") {
        next(false); // 拦截
    } else {
        next();      // 继续
    }
});

app.mount("#app");
